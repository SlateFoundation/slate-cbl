Role('Siesta.Util.Role.CanParseOs', {
    
    methods : { 
        
        parseOS : function (platform) {
            if (/Win/.test(platform)) return "Windows"
            
            if (/Mac/.test(platform)) return "MacOS"
            
            if (/Linux/.test(platform)) return "Linux"
            
            return "unknown"
        }
    }
});
